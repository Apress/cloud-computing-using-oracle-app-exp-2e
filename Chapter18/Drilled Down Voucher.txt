select
TD.LINE_NO,
TD.TRAN_NO,
TD.COCODE,
TD.COACODE,
TD.CCCODE,
TD.VCHDESCRIPTION,
TD.VCHDR,
TD.VCHCR,
TD.VCHREFERENCE,
TD.RECONCILED
from GL_TRAN_DETAIL TD, GL_TRAN_MASTER TM
where TD.TRAN_NO = TM.TRAN_NO and 
TM.VCHDATE = TO_DATE(:P44_VCHDATE,'DD-MON-YYYY') and
TM.VCHCODE = TO_NUMBER(:P44_VCHCODE) and
TM.VCHNO = TO_NUMBER(:P44_VCHNO) and
TM.COCODE = (select cocode from gl_users where upper(userid)=upper(:APP_USER))
